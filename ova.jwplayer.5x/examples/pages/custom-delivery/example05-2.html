Example - Impression serving empty VAST response - duration-less

This example illustrates how an ad server configuration can be set to ensure that the impression URL is called even if the VAST response provides no ad.


The configuration for this example is:


"" is defined as: 


<config>
   <ova.title>
       Example - Impression serving on an empty Ad VAST response
   </ova.title>
   <ova.json>
       {
            "debug": {
              "levels": "fatal, config, vast_template, vpaid, http_calls, playlist, api"
            },

            "shows": {
                "baseURL": "http://streaming.openvideoads.org/shows",
                "streams": [
                    { "file":"the-black-hole.mp4" },
                    { "file":"the-black-hole.mp4" },
                    { "file":"the-black-hole.mp4" },
                    { "file":"the-black-hole.mp4" }
                ]
            },

            "ads": {
                "schedule": [
                    {
                      "position": "pre-roll",
                      "applyToParts": [0,2],
                      "server": {
                          "type": "direct",
                          "forceImpressionServing": true,
                          "apiAddress": "../../../dist/templates/error-responses/vast1.0/empty-ad-vast-response.xml"
                      }
                    },
                    {
                      "position": "pre-roll",
                      "applyToParts": [1],
                      "server": {
                          "type": "direct",
                          "apiAddress": "../../../dist/templates/error-responses/vast1.0/empty-ad-vast-response.xml"
                      }
                    },
                    {
                      "position": "pre-roll",
                      "applyToParts": [3],
                      "server": {
                          "type": "direct",
                          "forceImpressionServing": true,
                          "apiAddress": "http://openx.openvideoads.org/openx/www...format=vast&charset=UTF-8"
                      }
                    }
                ]
            }
       }
   </ova.json>
</config>
