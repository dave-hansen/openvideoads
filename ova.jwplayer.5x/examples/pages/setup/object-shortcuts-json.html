Using "ova.json" flashvar


The configuration for this example is: 


<OBJECT classid="clsid:D27CDB6E-AE6D-11cf-96B8-444553540000"

 WIDTH="450" HEIGHT="300" id="player">
 <PARAM NAME=movie VALUE="player.swf">
 <PARAM NAME=quality VALUE=high>
 <PARAM NAME=bgcolor VALUE=#000000>
 <PARAM NAME=allowfullscreen VALUE="true">
 <PARAM NAME=allowscriptaccess VALUE="always">
 <PARAM NAME=flashvars VALUE="plugins=ova-jw">
<EMBED
 id="player"
 data="player.swf"
 src="player.swf"

 flashvars="plugins=ova-jw&file=the-black-hole.mp4&ova.json=%7B%22ads%22%3A%20%7B%20%22schedule%22%3A%20%5B%20%..."
>
</EMBED>
</OBJECT>
