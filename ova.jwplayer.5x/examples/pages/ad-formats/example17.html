Example - Multiple Pre-rolls and Overlays

This example illustrates the delivery of 2 pre-rolls, one after each other, followed by two overlays, one after each other on the show stream (the first overlay at 1 second into the show for 15 seconds, followed by a second at 20 seconds for 10 seconds).



The configuration for this example is:

"" is defined as: 


<config>
   <ova.title>
       Example - Multiple pre-rolls followed by muliple overlays during the show stream
   </ova.title>
   <ova.json>
       {
            "debug": {
                "levels": "fatal, config, vast_template, vpaid, http_calls, playlist, api"
            },

            "ads": {
                "notice": { "show": false },
                "servers": [
                    {
                       "type": "direct",
                       "apiAddress": "http://openx.openvideoads.org/openx/www/delivery/fc.php",
                       "allowAdRepetition": true
                    }
                ],
                "schedule": [
                    {
                       "zone": "5",
                       "position": "pre-roll"
                    },
                    {
                       "zone": "5",
                       "position": "pre-roll"
                    },
                    {
                       "zone": "30",
                       "startTime": "00:00:01",
                       "duration": 15,
                       "width": 450,
                       "height": 50
                    },
                    {
                       "zone": "30",
                       "startTime": "00:00:20",
                       "duration": 10,
                       "width": 450,
                       "height": 50
                    }
                ]
            }
       }
   </ova.json>
</config>
