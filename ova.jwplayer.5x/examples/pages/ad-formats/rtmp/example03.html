Example - Pre-roll and Overlay

A 30 second pre-roll ad plays followed by the show stream. 5 seconds into the show stream an overlay appears for 15 seconds - click on the overlay to start up a video ad. Once the video ad has played, the show stream resumes at the point at which the overlay was clicked.



The configuration for this example is:

"" is defined as: 


<config>
   <ova.title>
       Example - A pre-roll and show stream with an overlay ad
   </ova.title>
   <ova.json>
       {
            "regions": {
               "declarations": [
                   {
                      "id": "bottom",
                      "verticalAlign": "bottom",
                      "horizontalAlign": "center",
                      "backgroundColor": "#000000",
                      "padding": "-10 -10 -10 -10",
                      "width": 450,
                      "height": 50
                   }
                ]
            },

            "ads": {
                "servers": [
                    {
                       "type": "OpenX",
                       "apiAddress": "http://openx.openvideoads.org/openx/www/delivery/fc.php"
                    }
                ],

                "schedule": [
                    {
                       "zone": "1",
                       "position": "pre-roll"
                    },
                    {
                       "zone": "37",
                       "position": "bottom",
                       "width": 450,
                       "height": 50,
                       "startTime": "00:00:05",
                       "duration": "15"
                    }
                ]
            },

            "debug": {
                "debugger": "firebug"
            }
       }
   </ova.json>
</config>
