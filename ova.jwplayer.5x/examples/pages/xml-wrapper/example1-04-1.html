Example 1-04-1 - Single VAST 2 Wrapper to VAST 1 Non-Linear Ad

This example illustrates a VAST2 wrapper in action - a single layer of wrapper without ad templates.

300x250 Companion

The configuration for this example is:

"" is defined as: 


<config>
   <ova.title>
       Example 1-04-1 - Single VAST 2.0 Wrapper to VAST 1 Non-Linear Ad
   </ova.title>
   <ova.json>
       {
            "shows": {
                "streams": [
                    {
                        "file": "http://streaming.openvideoads.org/shows/the-black-hole.mp4",
                        "duration":"00:00:20"
                    }
                ]
            },

            "ads": {
               "companions": {
                   "regions": [
                      { "id":"companion-300x250", "width":"300", "height":"250" }
                   ]
               },
               "schedule": [
                  {
                      "position": "auto:bottom",
                      "startTime": "00:00:05",
                      "duration": "10",
                      "tag": "../../../dist/templates/wrapper/vast2-wrapper-to-vast1-nl-ad.xml"
                  }
               ]
            },

            "debug": {
                "levels": "fatal, config, vast_template, vpaid, http_calls, playlist, api"
            }
       }
   </ova.json>
</config>
