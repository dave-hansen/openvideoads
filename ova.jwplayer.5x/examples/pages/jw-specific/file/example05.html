Example 5 - Using SMIL playlists via the JW Flashvars

This example illustrates how to configure a SMIL playlist using the standard JW Player flashvars config.

The configuration for this example is:

"" is defined as: 


<config>
   <ova.title>
       Using SMIL playlists via the JW Flashvars
   </ova.title>
   <ova.json>
       {
            "blockUntilOriginalPlaylistLoaded": true,
            
            "debug": {
                "levels": "fatal, config, vast_template, vpaid, http_calls, playlist, api"
            },

            "ads": {
                "pauseOnClickThrough": true,
                "servers": [
                    {
                      "type": "OpenX",
                      "apiAddress": "http://openx.openvideoads.org/openx/www/delivery/fc.php"
                    }
                ],
                "schedule": [
                    {
                      "zone": "5",
                      "position": "pre-roll"
                    }
                ]
            }       
       }
   </ova.json>
</config>
