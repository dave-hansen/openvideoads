The configuration for this example is: 


<script type="text/javascript">
function dumpAdDescriptor() {
	var activeAdDescriptor = jwplayer('container').getPlugin('ova').getActiveAdDescriptor();
	if(activeAdDescriptor != null) {
		console.log(">>> ACTIVE AD DESCRIPTOR DUMP:");
		console.log(activeAdDescriptor);
	}
	else console.log(">>> NO DESCRIPTOR AVAILABLE");
}

jwplayer("container").setup({
       flashplayer: "", 

       playlist: [
          { 
              file: "http://streaming.openvideoads.org/shows/the-black-hole.mp4",
              duration: 10
          },
          { 
              file: "http://streaming.openvideoads.org/shows/the-black-hole.mp4",
              duration: 11
          }
       ],

       width: 650,
       height: 240,

       controlbar: "bottom",

       plugins: {
           "../../../dist/swf/ova-jw.js": { 
               "ads": {
                   "schedule": [
                       {
                           "position": "pre-roll",
                           "tag": "../../../dist/templates/vpaid/vast2-linear.xml"
                       }
                   ]
               },

               "debug": {
                  "levels": "fatal, config, vast_template, vpaid, http_calls, playlist, api"
               }
           }
       }
});
</script>
