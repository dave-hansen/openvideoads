Example 6 - A mid-roll and 2 overlays on a show with duration and a start time

Timings of the mid-roll and overlay are absolute not relative to the start of the show stream.

The configuration for this example is:

"" is defined as: 


<config>
   <ova.title>
       Example 6 - A mid-roll and 2 overlays on a show with duration and a start time
   </ova.title>
   <ova.json>
       {
            "shows": {
              "streams": [
                   {
                       "file": "OVA_PSEUDO_SHOW_STREAM_1",
                       "startTime": "00:00:15"
                       "duration": "00:01:30"
              	   }
              ]
            },

            "ads": {
              "servers": [
                   {
                      "type": "OpenX",
                      "apiAddress": "OVA_OPENX_API",
                      "allowAdRepetition": true
                   }
              ],
              "schedule": [
                  {
                      "zone": "5",
                      "position": "mid-roll",
                      "startTime": "00:00:35"
                  },
                  {
                      "zone": "33",
                      "position": "auto:bottom",
                      "width": 450,
                      "height": 50,
                      "startTime": "00:00:40",
                      "duration": "15"
                  },
                  {
                      "zone": "33",
                      "position": "auto:bottom",
                      "width": 450,
                      "height": 50,
                      "startTime": "00:01:05",
                      "duration": "15"
                  }
              ]
            },

            "debug": {
              "levels": "OVA_DEBUG",
              "debugger": "firebug"
            }
       }
   </ova.json>
</config>
