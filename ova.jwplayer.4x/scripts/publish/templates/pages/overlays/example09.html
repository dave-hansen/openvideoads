Example 9 - Enabling the standard cross-hair close button

The configuration for this example is:

"" is defined as: 


<config>
   <ova.title>
       Example 9 - Enabling the standard cross-hair close button
   </ova.title>
   <ova.json>
        {
            "debug": {
                "debugger": "firebug",
                "levels": "OVA_DEBUG"

            },
		
            "ads": {
                "servers": [
                   {
                      "type": "OpenX",
                      "apiAddress": "OVA_OPENX_API"
                   }
                ],
                "pauseOnClickThrough": true,
                "schedule": [
                   {
                      "zone": "28",
                      "startTime": "00:00:05",
                      "position": "auto:bottom",
                      "duration": "recommended:10"
                    }
	            ]
	        }
		}
   </ova.json>
</config>
