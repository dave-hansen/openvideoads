Example 18 - A mid-roll 10 seconds into a duration-less show stream

A mid-roll will play 10 seconds into the show stream.


The configuration for this example is:

"" is defined as: 


<config>
   <ova.title>
       Example 18 - A mid-roll 10 seconds into a duration-less show stream
   </ova.title>
   <ova.json>
       {
          "debug": {
              "debugger": "firebug",
              "levels": "OVA_DEBUG"
          },

          "ads": {
             "playOnce": false,
             "servers": [
                   {
                      "type": "OpenX",
                      "apiAddress": "OVA_OPENX_API"
                   }
             ],
             "schedule": [
                {
                   "zone": "5",
                   "position": "mid-roll",
                   "startTime": "00:00:10"
                }
             ]
          }
       }
   </ova.json>
</config>
