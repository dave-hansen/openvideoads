Example 14 - 2 stream with a post-roll on the second stream

2 streams with a mid-rolls 5 seconds into each stream = 4 streams

The configuration for this example is:

"" is defined as: 


<config>
   <ova.title>
       2 stream with a post-roll on the second stream
   </ova.title>
   <ova.json>
       {
           "shows": {
              "baseURL": "http://streaming.openvideoads.org/shows",
              "streams": [
                   { "file":"the-black-hole.mp4" },
                   { "file":"the-black-hole.mp4" }
              ]
           },

           "ads": {
              "servers": [
                  {
                    "type": "OpenX",
                    "apiAddress": "http://openx.openvideoads.org/openx/www/delivery/fc.php",
                    "allowAdRepetition": true
                  }
              ],
              "schedule": [
                  {
                    "zone": "19",
                    "position": "post-roll",
                    "applyToParts": [1]
                  }
              ]
           },

           "debug": {
              "debugger": "firebug",
              "levels": "fatal, config, vast_template, vpaid, playlist, api, analytics, http_calls"
           }
       }
   </ova.json>
</config>
