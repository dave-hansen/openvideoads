Example 4 - Modifying the Linear Ad Clickthrough Message

You can modify the font style, size and whether or not the message is shown.

The configuration for this example is:

"" is defined as: 


<config>
   <ova.title>
       Example 4 - Modifying the Linear Ad Clickthrough Message
   </ova.title>
   <ova.json>
       {
            "debug": {
              "debugger": "firebug",
              "levels": "fatal, config, vast_template, vpaid, playlist, api, analytics, http_calls"
            },

            "ads": {
                "pauseOnClickThrough": true,
                "clickSign": {
                   "enabled": true,
                   "verticalAlign": "center",
                   "horizontalAlign": "center",
                   "width": 150,
                   "height": 32,
                   "opacity": 0.5,
                   "borderRadius": 20,
                   "backgroundColor": "#000000",
                   "style": ".smalltext { font-size:12; }",
                   "html": "<p class=\"smalltext\" align=\"center\">CLICK ME!</p>",
                   "scaleRate": 0.75
                },
                "notice": { "show": true },
                "servers": [
                   {
                       "type": "OpenX",
                       "apiAddress": "http://openx.openvideoads.org/openx/www/delivery/fc.php",
                       "customProperties": {
                           "target": "category=food"
                       }
                    }
                ],
                "schedule": [
                   {
                      "zone": "1",
                      "position": "pre-roll"
                   }
                ]
            }
       }
   </ova.json>
</config>
