Example 16 - Ad runtime ad tag failover (on Ad stream error)


The configuration for this example is: 


flowplayer("a.example", "", {
    playlist: [
        { 
            url: "", 
            duration: 60 
        }
    ],
    
    plugins: {
        controls: {
            autoHide: "always"
        },

        ova: {
            url: "",

            "debug": {
                "levels": ""
            },

            "ads": {
                "schedule": [
                    {
                        "position": "pre-roll",
                        "loadOnDemand": true,
                        "server": {
                           "tag": "../../../dist/templates/error-responses/vast1.0/bad-stream-url.xml",
                           "failoverConditions": {
                               "onVPAIDError": {
                                  "always": true
                               },
                               "onStreamError": {
                                  "always": true
                               }
                           },
                           "failoverServers": [
                               {
                                  "tag": "http://openx.openvideoads.org/..."
                               }
                           ]
                        }
                    }
                ]
            }
         }
     }
});
